data_yield <-
  data_frame(
    ExptID = rep(letters[1:3], each = 10),
    Water = rep(c(0.1, 1), times = 15),
    Fertilizer = rep(1:10, times = 3),
    Yield = Fertilizer * 5 +  Water * 10 + rnorm(30, 0, 1)
    )
